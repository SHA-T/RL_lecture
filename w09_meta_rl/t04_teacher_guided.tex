\begin{frame}[c]{The Teacher}
	\begin{itemize}
		\item The teacher observes the student's reward $x_t$
		\item The action space consists of all instances $i$
		\item The teacher's reward is the change in the agent's reward:
			$$ r_t = x_t - x_{t-1}$$
	\end{itemize}
	
\end{frame}

\begin{frame}[c]{Variations on the idea}

	\begin{itemize}
		\item The same method can be applied to continuously parameterized environments \lit{\href{https://arxiv.org/pdf/1910.07224.pdf}{Portelas et al., 2019}}
		\item The teacher can be a fail-safe in safety-critical applications \lit{\href{https://papers.nips.cc/paper/2020/file/8df6a65941e4c9da40a4fb899de65c55-Paper.pdf}{Turchetta et al., 2020}}
		\item Guided Policy Search \lit{\href{https://graphics.stanford.edu/projects/gpspaper/gps_full.pdf}{Levine \& Koltun, 2013}} uses an expert policy to sample trajectories (not necessarily across instances)	
	\end{itemize}

\end{frame}
